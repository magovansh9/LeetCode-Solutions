#include <stack>
using namespace std;
class Solution {
public:
    bool isValid(string s) {
    
        stack<char> stack;
        if(s.length()==1)
            return false;
        
        if(s.at(0)==')' || s.at(0)==']' || s.at(0)=='}')
            return false;
        
        for(int i=0;i<s.length();i++)
        {
            if(s.at(i)=='(' || s.at(i)=='[' || s.at(i)=='{')
            {
                stack.push(s.at(i));
            }
            
            else if(s.at(i)==')') 
            {
            if(stack.size()==0)
            {
                return false;
            }   
            else {
                if(stack.top()=='(')
                stack.pop(); 
                
                else return false;
                 }
            }
            
            else if(s.at(i)==']') 
            {
            if(stack.size()==0)
            {
                return false;
            }   
            else {
                if(stack.top()=='[')
                stack.pop(); 
                
                else return false;
                 }
            }
            
            else if(s.at(i)=='}') 
            {
            if(stack.size()==0)
            {
                return false;
            }   
            else {
                if(stack.top()=='{')
                stack.pop(); 
                
                else return false;
                 }
            }
            
        }
        
        return (stack.size()==0) ? true : false;
       
    }

};
