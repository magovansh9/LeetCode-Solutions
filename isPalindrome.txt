#include <string>
class Solution {
public:
    bool isPalindrome(int x) {
        
        string a = to_string(x);
        
        if(a.at(0) == '-')
        {
            return false;
        }
        
        for(int i=0; i< a.length(); i++)
        {
            if(a.at(i)!=a.at(a.length()-i-1))
            {
                return false;
            }
        }
        
        return true;
        
    }
};
