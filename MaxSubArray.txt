class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        
  int n=nums.size();
        int maxe=nums[0],main_max=nums[0];
        for(int i=1;i<n;i++)
        {
            maxe=max(maxe+nums[i],nums[i]);
            if(main_max<maxe)
                main_max=maxe;
        }
        return main_max;
    }
};
